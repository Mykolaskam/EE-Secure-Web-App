<?php 
  require_once '../src/UserModel.php';
  require_once '../src/SQLWrapper.php';
  use PHPUnit\Framework\Testcase;

  final class ProfileRetrievealTest extends Testcase {

    public function testProfileRetrieval(){
      $arr_account_details = array('MatthewBaber', 'ABC', 'Matthew', 'Baber');
      $user_obj  = new UserModel();
      $user_obj->set_user_values($arr_account_details[0], $arr_account_details[1], $arr_account_details[2], $arr_account_details[3]);
      $sql_wrapper = new SQLWrapper();
      $sql_wrapper->create_user_var($user_obj)
      $this->asserEquals($arr_account_details[0], $user_obj->('username'));
    }
  }
 ?>
